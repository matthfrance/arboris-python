>>> np.dtype([('a','f8'),('b','S10')]) 

dtype([('a', '<f8'), ('b', '|S10')])

dtype([('f0', '<i4'), ('f1', '<f8', (2, 3))])

dtype([('gender', '|S1'), ('age', '|u1')])
